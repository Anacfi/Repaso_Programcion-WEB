            input type="button" value="MOSTRAR" onclick="mostrar();" -- Se crea un boton llamando a una funcion
            input type="button" value="OCULTAR" onclick="ocultar();" -- "lo mismo"

                $("#texto").hide();

            function cambiarparrafo(){

                if(text == ""){
                    $("#DivAlert").show();
                }
                else{
                   $("#texto").html(text); 
                }
            }
